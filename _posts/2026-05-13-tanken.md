<script>
var elem = function(name) {
  return document.getElementById(name);
}
var set = function(name, value) {
  elem(name).innerHTML = value;
};
var handleOkResponse = function(response) {
  if (response.ok) {
    elem('preis-ok').style.display='initial';
    set('brand', response.station.brand);
    set('street', response.station.street);
    set('houseNumber', response.station.houseNumber);
    set('postCode', response.station.postCode);
    set('place', response.station.place);
    set('dieselpreis', response.station.diesel);
    set('e5', response.station.e5);
    set('e10', response.station.e10);
  } else {
    elem('preis-nok').style.display='initial';
    elem('error-message').innerHTML = response.message;
  }
};
var handleNokResponse = function(response) {
  elem('preis-nok').style.display='initial';
};
var getData = function() {
  var url = "../assets/js/detail.json";
  var xhr = new XMLHttpRequest();
  xhr.open('GET', url, true);
  xhr.responseType = 'json';
  xhr.onload = function() {
    var status = xhr.status;
    if (status === 200) {
      handleOkResponse(xhr.response);
    } else {
      handleNokResponse(xhr.response);
    }
  };
  xhr.send();
};
var getTime = function() {
  var url = "../assets/js/detail_time.json";
  var xhr = new XMLHttpRequest();
  xhr.open('GET', url, true);
  xhr.responseType = 'json';
  xhr.onload = function() {
    var status = xhr.status;
    if (status === 200) {
      var d = new Date(xhr.response.last_update);
      set('last_update', new Intl.DateTimeFormat("de-DE", {dateStyle: "full", timeStyle: "short"}).format(d));
    }
  };
  xhr.send();
};
getData();
getTime();
</script>
